practical example of using I2C_Slave_Device class for communication with SFT02 ultra sound range finder

#ifndef SRF02_H_
#define SRF02_H_
#include <stdint.h>
#include "i2c_slave_device.h"
#include "setila_errors.h"
#define SRF02_RANGING_IN_PROGRESS 0xFF
enum class SRF02_Register: uint8_t {
COMMAND = 0x00,
UNUSED,
RANGE_HIGH_BYTE,
RANGE_LOW_BYTE,
AUTOTUNE_MINIMUM_HIGH_BYTE,
AUTOTUNE_MINIMUM_LOW_BYTE
};
enum class SRF02_Command: uint8_t {
REAL_RANGING_MODE_inch = 0x50,
REAL_RANGING_MODE_cm = 0x51,
REAL_RANGING_MODE_us = 0x52,
FAKE_RANGING_MODE_inch = 0x56,
FAKE_RANGING_MODE_cm = 0x57,
FAKE_RANGING_MODE_us = 0x58,
PING_NO_RANGING = 0x5C,
FORCE_AUTOTUNE = 0x60,
CHANGE_I2C_ADDR_STEP1 = 0xA0,
CHANGE_I2C_ADDR_STEP2 = 0xA5,
CHANGE_I2C_ADDR_STEP3 = 0xAA
};
typedef enum {
SRF02_READ_WRITE_ERROR = -5,
SRF02_UNSUPPORTED_COMMAND = -4,
SRF02_OBJECT_FOUND = -3,
SRF02_NO_OBJECT_IN_RANGE = -2,
SRF02_READING_RANGE_FAILED = -1,
SRF02_OPERATION_SUCCESSFUL = 0
} srf02_status_t;
class SRF02 : public I2C_Slave_Device
{
private:
uint8_t m_software_version = 0x00;
uint16_t m_distance = 0x00;
public:
SRF02(): I2C_Slave_Device(0x70) {};
explicit SRF02(uint8_t i2c_slave_address): I2C_Slave_Device(i2c_slave_address) {};
~SRF02() {};
const uint16_t distance(void) const { return m_distance; };
int srf02_find_object(SRF02_Command measurement_unit);
int srf02_find_object(SRF02_Command measurement_unit, uint16_t *distance);
srf02_status_t srf02_set_new_i2c_address(uint8_t new_i2c_address);
const char *srf02_status_to_string(srf02_status_t srf02_status);
};
#endif /* SRF02_H_ */
